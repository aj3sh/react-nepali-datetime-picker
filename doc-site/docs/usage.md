# Usage



```js
import {
  DatePicker,
  TimePicker,
  DateTimePicker,
  StaticCalendar,
  StaticDesktopTime,
} from 'react-nepali-datetime-picker';

import 'react-nepali-datetime-picker/dist/style.css';

const MyComponent = () => {
  return (
    <>
      <DatePicker />
      <TimePicker />
      <DateTimePicker />
      <StaticCalendar />
      <StaticDesktopTime/>
    </>
  );
};

export default MyComponent;
```
